数据库t01
    用户表
    id          int        主键      自动增长     编号
    account     varchar    允许空                账号
    password    varchar    允许空                密码
    name        varchar    允许空                姓名

create table users
(
id int auto_increment,
account varchar(20) null,
password varchar(20) null,
name varchar(20) null,
constraint users_pk
primary key (id)
)
comment '用户表';

用户账号是唯一的
alter table users
	add constraint users_uni
		unique (account);
